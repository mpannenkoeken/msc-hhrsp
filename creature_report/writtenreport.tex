\subsection{Home Healthcare}
Home Healthcare is the process by which certain health-related services are provided to patients in their home. These services typically involve nursing care, though can also consist of prescription and other medical supply delivery as well as assistance with day-to-day activity \cite{al-hamad}. These services provided by home healthcare models are becoming increasingly essential as demographic changes yield older populations, greater geographical dispersion relative to both family and central hospital campuses, and a general preference toward aging in place. Additionally, home healthcare services can prove essential to modern healthcare systems by enabling personalized, expert care for patients with chronic illness, mobility challenges, and post-op recovery needs who may not otherwise be able to receive such quality care \cite{atta25, Mankowska2014}.

Consequently, the importance of home healthcare companies, i.e. companies providing these services, is of increasing social interest. These groups must meet several operational challenges to meet the needs of their client base, and must do so with efficiency to ensure competitive advantage in a market system. Among the challenges of these groups is the need to find ways to efficiently provide diverse services requiring differing numbers and types of caregivers \cite{begur97, Mankowska2014}. Not only must these services be provided by varying numbers of caregivers with various qualification, they must also be delivered within strict time windows according to patient care needs \cite{begur97}. while meeting regulatory compliance requirements. Given the operational complexity and social importance of these operations, the operations of home healthcare companies are a domain with high-potential for operations research applications.

\begin{modelsets}
	$D$ & set of days to schedule caregivers\\
	$C^d$ & set of caregivers available on day $d$\\
	$C_D^d \subseteq C^d$ & set of driving caregivers availabl on day $d$\\
	$L$ & set of localities served
\end{modelsets}

\begin{modelparams}
	$d_{ij}$ & \quad distance or drive time between caregivers $i$ and $j$\\
	$r_{jl}$ & \quad distance or drive time between caregiver $j$ and locality $l$\\
	$F_{ij}^d$ & \quad number of pair visits on day $d$ where both caregivers $i$ and $j$ are familiar to the client\\
	$f_{ij}^d$ & \quad number of pair visits on day $d$ where either caregiver $i$ or $j$ are familiar to the client\\
	$s_i^d$ & \quad number of solo visits on day $d$ where caregiver $i$ is familiar to the client\\
	$V^d$ & \quad total visit minutes on day $d$\\
	$V_{Pl}^d$ & \quad visit minutes in locality $l$ on day $d$ requiring a caregiving pair\\
	$V_{sl}^d$ & \quad visit minutes in locality $l$ on day $d$ requiring a single caregiver\\
	$l_i$ & \quad binary indicator whether caregiver $i$ is located in locality $l$\\
	$K$ & \quad maximum distance carers will travel for pairing
\end{modelparams}

\begin{modelvars}
	$v_{il}^d$ & \quad variable representing the linearization of the product of binary variables $z_{il}^d y_{i}^d$\\
	$w_{ijl}^d$ & \quad variable representing the linearization of the product of binary variables $z_{jl}^d x_{ij}^d$
\end{modelvars}
\begin{table}[H]
	\begin{tabular}{l}
		$x_{ij}^d = 
			\begin{cases}
				1 & \text{if caregiver } i \in C_D^d \text{paired with caregiver } j \in C^d \setminus \{i\} \text{on day } d\\
				0 & \text{else}
			\end{cases}
		$ \\
		$y_{j}^d =
			\begin{cases}
				1 & \text{if caregiver } j \in C^d \text{is unpaired (i.e. does solo visits) on day } d\\
				0 & \text{else}
			\end{cases}
		$ \\
		$z_{il}^d = 
			\begin{cases}
				1 & \text{if caregiver } i \in C^d \text{assigned to locality } l \in L \text{on day } d\\
				0 & \text{else}
			\end{cases}
		$
	\end{tabular}
\end{table}

\begin{alignat}{4}
	\max \quad 
	& \sum\nolimits_{d \in D} \sum\nolimits_{j\in C^d} \left( s_j y_j^d + \sum\nolimits_{i \in C_D^d \setminus \{j\}} \left( F_{ij}^d x_{ij}^d + f_{ij}^d x_{ij}^d \right)\right) 
	&&&&&& \label{obj1} \\
	\min \quad
	& \sum\nolimits_{d \in D} \sum\nolimits_{j\in C^d} \sum\nolimits_{i \in C_D^d \setminus \{j\}} d_{ij} x_{ij}^d + \sum\nolimits_{l \in L} r_{jl} \left( x_{ij}^d + y_j^d \right) 
	&&&&&& \label{obj2} \\
	\text{s.t.}\quad 
	& y_j^d + \sum\nolimits_{i \in C_D^d} x_{ij}^d & = & 1
	& \forall j \in C^d \setminus C_D^d , \forall d \in D \label{assignWalkerOnce} \\
	& y_j^d + \sum\nolimits_{i \in C_D^d \setminus \{j \}} x_{ij}^d + \sum\nolimits_{k \in C_D^d \setminus \{j\}} x_{jk}^d & = & 1
	& \forall j \in C_D^d , \forall d \in D \label{assignDriverOnce} \\
	& \sum\nolimits_{i \in C_D^d} \sum\nolimits_{j \in C^d \setminus \{ i \}} l_j x_{ij}^d + w_{ijl}^d - \sum\nolimits_{l' \in L \setminus \{l\}} w_{ijl'}^d & \leq & \lceil |C^d| \tfrac{V_{Pl}^d}{V^d} \rceil + \epsilon
	& \forall d \in D , \forall l \in L \label{upperPairLocale} \\
	& \sum\nolimits_{i \in C_D^d} \sum\nolimits_{j \in C^d \setminus \{ i \}} l_j x_{ij}^d + w_{ijl}^d - \sum\nolimits_{l' \in L \setminus \{l\}} w_{ijl'}^d
	& \geq & \lfloor \tfrac{|C^d|}{2} \tfrac{V_{Pl}^d}{V^d} \rfloor - \epsilon
	& \forall d \in D , \forall l \in L \label{lowerPairLocale} \\
	& \sum\nolimits_{j \in C^d \setminus C_D^d} \left( l_j y_j^d + \sum\nolimits_{i\in C_D^d} \left( l_i y_i^d + v_{il}^d - \sum\nolimits_{l' \in L \setminus \{l\}} v_{il'}^d \right) \right) & \leq & \lceil |C^d| \tfrac{V_{sl}^d}{V^d} \rceil + \epsilon
	& \forall d \in D, \forall l \in L \label{upperSoloLocale} \\
	& \sum\nolimits_{j \in C^d \setminus C_D^d} \left( l_j y_j^d + \sum\nolimits_{i\in C_D^d} \left( l_i y_i^d + v_{il}^d - \sum\nolimits_{l' \in L \setminus \{l\}} v_{il'}^d \right) \right) & \geq & \lfloor \tfrac{|C^d|}{2} \tfrac{V_{sl}^d}{V^d} \rfloor - \epsilon
	& \forall d \in D, \forall l \in L \label{lowerSoloLocale} \\
	& x_{ij}^d & = & 0 
	& \forall i \in C_D^d \forall j \in \{\, j \in C^d \mid d_{ij} \ge K \,\} \label{pairDistLimit}\\
	& w_{ijl}^d & \geq & 0 
	& \forall i \in C_D^d, \forall j \in C^d \setminus \{i\}, \forall l \in L, \forall d \in D \label{wLin1} \\
	& w_{ijl}^d & \leq & x_{ij}^d
	& \forall i \in C_D^d, \forall j \in C^d \setminus \{i\}, \forall l \in L, \forall d \in D \label{wLin2} \\
	& w_{ijl}^d & \leq & z_{jl}^d
	&  \forall i \in C_D^d, \forall j \in C^d \setminus \{i\}, \forall l \in L, \forall d \in D \label{wLin3} \\
	& z_{jl}^d + x_{ij}^d - 1 & \leq & w_{ijl}^d
	&  \forall i \in C_D^d, \forall j \in C^d \setminus \{i\}, \forall l \in L, \forall d \in D \label{wLin4} \\
	& v_{il}^d & \geq & 0
	& \forall i \in C^d, \forall l \in L, \forall d \in D \label{vLin1} \\
	& v_{il}^d & \leq & y_{i}^d
	& \forall i \in C^d, \forall l \in L, \forall d \in D \label{vLin2} \\
	& v_{il}^d & \leq & z_{il}^d
	& \forall i \in C^d, \forall l \in L, \forall d \in D \label{vLin3} \\
	& z_{jl}^d + y_{i}^d - 1 & \leq & v_{il}^d
	& \forall i \in C^d, \forall l \in L, \forall d \in D \label{vLin4} \\
	& x_{ij}^d & \in & \{0,1\}
	& \forall i \in C_D^d, \forall j \in C^d \setminus \{i\}, \forall d \in D \label{domX} \\
	& y_i^d & \in & \{0,1\}
	& \forall i \in C^d, \forall d \in D \label{domY} \\
	& z_{il}^d & \in & \{0,1\}
	& \forall i \in C^d, \forall l \in L, \forall d \in D \label{domZ}
\end{alignat}
